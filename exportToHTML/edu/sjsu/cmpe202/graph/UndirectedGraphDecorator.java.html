6    /** 
7     * Decorator for undirected graph. 
8     * 
9     * @author rwatsh on 8/14/16.

11   public class UndirectedGraphDecorator extends GraphDecorator { 
12       public UndirectedGraphDecorator(Graph g) { 
13           super(g); 
14       } 
15    
16       public List<Vertex> getVertexes() { 
17           return super.getVertexes(); 
18       } 
19    
20       /** 
21        * This method returns the modified (decorated) edges list. 
22        * 
23        * @return 
24        */ 
25       public List<Edge> getEdges() { 
26           List<Edge> edges = super.getEdges(); 
27           List<Edge> revEdges = new ArrayList<>(); 
28           // For each edge create a reverse edge 
29           for (Edge e: edges) { 
30               revEdges.add(new Edge("R" + e.getId(), e.getDestination(), e.getSource(), e.getWeightTime(), e.getWeightDistance())); 
31           } 
32           edges.addAll(revEdges); 
33           return edges; 
34       } 
35   }
