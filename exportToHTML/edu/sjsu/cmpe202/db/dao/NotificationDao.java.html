12   public class NotificationDao { 
13       public static void sendNotifications(Notification n) { 
14           String note = "INSERT into notification(notifyuser_id,date,message)" + 
15                   "VALUES(:notifyuser_id,:date,:message)"; 
16           try (Connection con = (new SQLConnection()).getConnection()) { 
17               con.createQuery(note) 
18                       .addParameter("notifyuser_id", n.getNotifyUserId()) 
19                       .addParameter("date", n.getDate()) 
20                       .addParameter("message", n.getMessage()) 
21                       .executeUpdate(); 
22           } 
23       } 
24    
25       public static List<Notification> getNotifications() { 
26           String fetchRouteMapSql = "SELECT * FROM notification"; 
27           List<Notification> notifyList; 
28    
29           try (Connection con = (new SQLConnection()).getConnection()) { 
30               notifyList = con.createQuery(fetchRouteMapSql) 
31                       .executeAndFetch(Notification.class); 
32           } 
33           return notifyList; 
34       }
